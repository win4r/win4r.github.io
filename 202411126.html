brew install uv


mkdir weather_service
cd weather_service

# 使用 uv init 初始化，添加 --no-workspace 参数
uv init --no-workspace

# 然后再运行 create-mcp-server
uvx create-mcp-server \
  --name weather_service \
  --version "0.1.0" \
  --description "A MCP server project for weather service"
