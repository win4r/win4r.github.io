import http.client
import json
import ssl

if __name__ == '__main__':
    # 创建一个忽略证书验证的 SSL 上下文
    context = ssl._create_unverified_context()
    conn = http.client.HTTPSConnection("cephalon.cloud", context=context)

    payload = json.dumps({
        "model": "Llama-3.1-70B-Instruct",
        "messages": [
            {
                "role": "system",
                "content": "You are a helpful assistant."
            },
            {
                "role": "user",
                "content": "Hello!"
            }
        ],
    })

    headers = {
        'Authorization': "Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJJbmZvIjoiYXBpOjE4NTUwMzg0MjE2MTY3NTQ2ODgiLCJleHAiOjE3NjU4NTE4NjB9.Te1MnmAASB2Kg5IgaWtga-7uZk-hwVBZyWyj83ALoGk",
        'Content-Type': 'application/json'
    }

    conn.request("POST", "/user-center/v1/model/chat/completions", payload, headers)
    res = conn.getresponse()
    data = res.read()
    
    # 将返回的数据解析为 JSON 格式
    response = json.loads(data.decode("utf-8"))
    
    # 提取 content 的内容并打印
    content = response["choices"][0]["message"]["content"]
    print(content)
